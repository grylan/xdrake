/* Desidero acquistare un cellulare usato. La cifra massima che voglio spendere è:
i. 100 euro come prezzo base.
ii. 40 euro aggiuntivi per ognuna delle caratteristiche a cui sono interessato.
iii. 20 euro in meno per ogni mese in cui il cellulare è stato posseduto dal precedente proprietario.*/
#include <stdio.h>
#include <stdlib.h>
/*Definisca tramite #define i valori che compongono il prezzo*/
#define BASE 100
#define FEATURE_COST 40
#define TIME_COST 20


int main()
{
    /*Definisca le variabili intere price, features, months e years */
    int price;
    int feature;
    int months;
    int years;

    /*Acquisisca da tastiera la quantità di caratteristiche possedute dal cellulare (features) e di anni per cui il cellulare è stato posseduto (years). */
    printf("Features: ");
    scanf("%d",&feature);
    printf("Years: ");
    scanf("%d",&years);

    /*Calcoli il numero di mesi per cui il cellulare è stato posseduto (months)*/
    months=years*12;

    /*Calcoli il prezzo massimo spendibile per il cellulare (price).*/
    price = BASE+FEATURE_COST*feature-TIME_COST*months;

    /*Stampare il risultato a video*/
    printf("\nThe maximum price is %d",price);



    return 0;
}
